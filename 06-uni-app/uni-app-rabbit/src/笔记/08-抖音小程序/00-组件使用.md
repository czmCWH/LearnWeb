```vue
  :lazy-load="true"
  :webp="true"
/>
```

# 2、movable-view 组件

如下代码所示使用 movable-view 组件进行缩放时，设置了 `scale-min="0.1` ，盒子无法缩放至0.1倍，这是为什么？

```vue
<movable-area style="width: 300px; height: 300px; background-color: pink;" scale-area>
  <movable-view style="width: 600px; height: 600px; background-color: red;" direction="all" bindscale="onScale" scale scale-min="0.1" scale-max="4" scale-value="{{scale}}">
     <image
      src="https://info.com/admin/1/2024/12/10/173381284970.jpg"
      mode="aspectFill"
      style="width: 100%; height: 100%;"
    />
  </movable-view>
</movable-area>

<script>
const app = getApp()

Page({
  data: {
    scale: 0.1
  },
  onLoad: function () {
    console.log('Welcome to Mini Code')
  },
  onScale: function (e: any) {
    console.log('--- e = ', e.detail.scale);
  }
})
</script>
```

官方文档上说明 scale-min 可以自定义配置，但是又介绍 scale-value 取值范围为 0.5 ~ 10，这和自定义配置 scale-min 自相矛盾？
由于 `scale-value` 最小为 0.5，那么加载图片的css大小为盒子显示的大小。对 movable-view 进行放大时，其内嵌的 image 组件显示会变模糊，因为  movable-view 的缩放是使用 transform 伪缩放，不是真正的改变盒子的大小。


# 3、uview-plus 组件库的 up-waterfall 组件不渲染数据，该如何解决？

vue3+ts+uni-app开发抖音小程序，自定义组件并传值如下代码所示：

const props = defineProps<{
  list: Object[]
}>()

const dataList = ref([{name: "12"}])
<AiHxTemplateList :list="dataList" />

解决：不要把 up-waterfall 组件封装到自定义组件里，应该直接在 page 中构建，否则会导致数据更新不渲染。
